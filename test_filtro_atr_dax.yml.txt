name: Test Filtro ATR DAX (causale + bootstrap)

on:
  workflow_dispatch: {}

jobs:
  test-filtro-atr-dax:
    runs-on: ubuntu-latest
    timeout-minutes: 20

    steps:
      - name: Checkout repo
        uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Installa dipendenze
        run: pip install pandas numpy requests dukascopy-python

      - name: Esegui test filtro ATR DAX
        env:
          CLOUDFLARE_ACCOUNT_ID: ${{ secrets.CLOUDFLARE_ACCOUNT_ID }}
          CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}
        run: python test_filtro_atr_dax.py
