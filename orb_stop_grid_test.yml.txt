name: ORB Stop Grid Test (train, ATR moltiplicatore)

on:
  workflow_dispatch: {}

jobs:
  orb-stop-grid:
    runs-on: ubuntu-latest
    timeout-minutes: 15

    steps:
      - name: Checkout repo
        uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Installa dipendenze
        run: pip install dukascopy-python pandas numpy

      - name: Esegui griglia stop ATR
        run: python orb_stop_grid_test.py

      - name: Carica risultati come artifact
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: orb-stop-grid-${{ github.run_id }}
          path: orb_stop_grid_train_results_2015_2016.csv
          retention-days: 30
          if-no-files-found: warn
