name: Asset Pair Comparison (DAX+FTSE100 vs DAX+GOLD vs FTSE100+GOLD)

# Confronta 3 coppie candidate per l'universo asset attivo, capitale
# 2.000€, criterio di promozione fissato in chat il 15/07/2026:
#   1) PnL/|drawdown| positivo in 5/5 periodi
#   2) rapporto aggregato >= baseline (DAX+FTSE100) + 10%
#   3) nessun periodo con drawdown singolo oltre -35%
# Job leggero: 3 coppie x 5 periodi, nessuna griglia/random baseline —
# solo backtest deterministico, ~5-10 minuti stimati.
on:
  workflow_dispatch: {}

jobs:
  asset-pair-comparison:
    runs-on: ubuntu-latest
    timeout-minutes: 30

    steps:
      - name: Checkout repo
        uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Installa dipendenze
        run: pip install pandas numpy requests

      - name: Scarica OHLC DAX+FTSE100+GOLD da Cloudflare D1
        env:
          CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}
          CLOUDFLARE_ACCOUNT_ID: ${{ secrets.CLOUDFLARE_ACCOUNT_ID }}
        run: python fetch_ohlc_pairs.py

      - name: Confronto 3 coppie (DAX+FTSE100 / DAX+GOLD / FTSE100+GOLD)
        run: python asset_pair_comparison.py

      - name: Carica risultati come artifact
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: asset-pair-comparison-${{ github.run_id }}
          path: results/
          retention-days: 30
          if-no-files-found: warn
